Après avoir créé le vnet et le vnet gateway, il faut créer le vpn P2S grace au info ci-dessous :

#Create the root cert

$cert = New-SelfSignedCertificate -Type Custom -KeySpec Signature -Subject "CN=PPUP2SRootCert" -KeyExportPolicy Exportable -HashAlgorithm sha256 -KeyLength 2048 -CertStoreLocation "Cert:\CurrentUser\My" -KeyUsageProperty Sign -KeyUsage CertSign

 

# Create Client Cert

New-SelfSignedCertificate -Type Custom -DnsName P2SChildCert -KeySpec Signature -Subject "CN=PPUP2SClientCert1" -KeyExportPolicy Exportable -HashAlgorithm sha256 -KeyLength 2048 -CertStoreLocation "Cert:\CurrentUser\My" -Signer $cert -TextExtension @("2.5.29.37={text}1.3.6.1.5.5.7.3.2")
