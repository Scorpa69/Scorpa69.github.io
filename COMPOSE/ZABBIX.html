Pour monitorer le serveur docker qui héberge zabbix, ne pas oublier de changer le listenport de l'agent zabbix installer sur le serveur docker et ajouter le user zabbix au groupe docker :

usermod -a -G docker zabbix

version: '3.5'

services:
  zbx-mysql:
    image: mysql:8.0
    container_name: zbx-mysql
    command:
    - mysqld
    - --character-set-server=utf8mb4
    - --collation-server=utf8mb4_bin
    restart: always
    environment:
      MYSQL_DATABASE: zabbix
      MYSQL_USER: zabbix

      - mysql_root_password

      - zbx_net

      - mysql_data:/var/lib/mysql

      - zbx_net
