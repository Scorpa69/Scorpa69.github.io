Fonctionnel :

version: '3.5'

services:
  db:
    image: percona:5.7.35
    container_name: db
    command: --default-authentication-plugin=mysql_native_password
    restart: always
    environment:
      MYSQL_ROOT_PASSWORD: /run/secrets/mysql_root_pass
      MYSQL_DATABASE: kanboard
      MYSQL_USER: kanboard
      MYSQL_PASSWORD: /run/secrets/mysql_pass
    volumes:
      - db:/var/lib/mysql
    networks:
      - kanboard-net
    ports:
      - "3306"
    secrets:
      - mysql_pass
      - mysql_root_pass

  kanboard:
    image: kanboard/kanboard:latest
    container_name: kanboard
    restart: always
    environment:
      #DB_DRIVER: mysql <--- attention fait planter l'app !!!
      DB_HOST: db
      DB_PORT: 3306
      DB_NAME: kanboard
      DB_USER: kanboard
      DB_PASSWORD: /run/secrets/mysql_pass
    ports:
      - "8085:80"
    depends_on:
      - db
    networks:
      - kanboard-net
    volumes:
      - kanboard_data:/var/www/app/data
      - kanboard_plugins:/var/www/app/plugins
      - kanboard_ssl:/var/www/app/ssl
    secrets:
      - mysql_pass

volumes:
  kanboard_data:
  kanboard_plugins:
  kanboard_ssl:
  db:

networks:
  kanboard-net:
    driver: bridge
    ipam:
      driver: default
      config:
        - subnet: 192.168.250.0/24    

secrets:
  mysql_pass:
    file: ./env/.MYSQL_PASS
  mysql_root_pass:
    file: ./env/.MYSQL_ROOT_PASS
