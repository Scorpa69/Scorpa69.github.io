Suppression fichiers et dossiers dans un répertoire donné :


Sans rétention

#!/bin/sh
for fu in user1 user2 user3 user4
do
find /home/"$fu"/ -mindepth 1 -type d -exec rm -rf {} \;
find /home/"$fu"/ -type f -exec rm {} \;
done

Avec rétention 4 heures

#!/bin/sh
for fu in user1 user2 user3 user4
do
find /home/"$fu"/ -mindepth 1 -type d -mmin +240 -exec rm -rf {} \;
find /home/"$fu"/ -type f -mmin +240 -exec rm {} \;
done

Avec rétention 8 jours

#!/bin/sh
for fu in user1 user2 user3 user4
do
find /home/"$fu"/ -mindepth 1 -type d -mtime +7 -exec rm -rf {} \;
find /home/"$fu"/ -type f -mtime +7 -exec rm {} \;
done
