Exemple IA zabbix-agent + mysql sur alpine : A tester !!!

# Use the Alpine base image
FROM alpine:latest

# Update the package repository and install required packages
RUN apk update && \
    apk add --no-cache zabbix-agent mysql-client

# Copy the Zabbix agent configuration file
COPY zabbix_agentd.conf /etc/zabbix/zabbix_agentd.conf

# Expose the Zabbix agent port
EXPOSE 10050

# Start the Zabbix agent
CMD ["zabbix_agentd", "-c", "/etc/zabbix/zabbix_agentd.conf", "-f"]
